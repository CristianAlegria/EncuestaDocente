


********************** MODULO GESTIONAR ENCUESTA***************************


CREATE TABLE ENCUESTA(
IDENCUESTA SERIAL NOT NULL,
FECHA date NOT NULL,
PERIODO CHAR(50),
CONSTRAINT PK_ENCUESTA PRIMARY KEY(IDENCUESTA));


CREATE TABLE ENCUESTA_ASIGNATURA(
IDENCUESTA_ASIGNATURA SERIAL NOT NULL,
IDENCUESTA int NOT NULL,
IDASIGNATURA char(10)  NOT NULL,
CONSTRAINT PK_ENCUESTA_ASIGNATURA PRIMARY KEY(IDENCUESTA_ASIGNATURA),
CONSTRAINT FK_ENCUESTA_ASIGNATURA_ENCUESTA FOREIGN KEY(IDENCUESTA)
REFERENCES ENCUESTA(IDENCUESTA),
CONSTRAINT FK_ENCUESTA_ASIGNATURA_ASIGNATURA FOREIGN KEY(IDASIGNATURA)
REFERENCES asignatura(IDASIGNATURA )

CREATE TABLE FORMULARIO(
IDFORMULARIO SERIAL NOT NULL,
IDENCUESTA_ASIGNATURA INT NOT NULL,
IDPREGUNTA int  NOT NULL,
ESTADO boolean NOT NULL,
FECHA date,
PERIODO CHAR(50),
CONSTRAINT PK_FORMULARIO PRIMARY KEY(IDFORMULARIO),
CONSTRAINT FK_FORMULARIO_PREGUNTA FOREIGN KEY(IDPREGUNTA)
REFERENCES PREGUNTA(IDPREGUNTA),
CONSTRAINT FK_FORMULARIO_ENCUESTA_ASIGNATURA FOREIGN KEY(IDENCUESTA_ASIGNATURA)
REFERENCES ENCUESTA_ASIGNATURA(IDENCUESTA_ASIGNATURA));
